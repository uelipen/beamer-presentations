  \frame{
    \frametitle{CHIME-absorber}
    \begin{itemize}
        \item CHIME surveys $0.8<z<2.4$
        \item forecast up to $10^5$ new blind detection systems, orders of
          magnitude improvement over existing surveys, different
          hemisphere/redshift range from ASKAP-FLASH (Yu++ 2014) 
        \item map galaxy evolution
        \item potentially measure cosmic acceleration through
          Sandage-Loeb effect
    \end{itemize}
}

  \frame{
\vspace{-0.5in}
    \frametitle{Technical Steps}
    \begin{itemize}
    \item absorption depth $\proto 1/T$, sensitive to cold gas
        \item narrow features, often finer than km/s
        \item requires $R\sim 10^5$
        \item rechannelize CHIME-FRB beam stream to 3kHz
        \item use earth-sun doppler motion to reject foregrounds, RFI
    \end{itemize}
  }


  \frame{

    \end{itemize}
